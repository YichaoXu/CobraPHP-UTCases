function resolveAfter1Seconds(val) {
    return new Promise(resolve => {
      setTimeout(() => {
        resolve(val);
      }, 1000);
    });
}
  
async function asyncCall(val) {
    var result = await resolveAfter1Seconds(val);
    res.writeHead(200, {"Content-Type" : "text/html"});
    res.write(result);     
    res.end();
}
